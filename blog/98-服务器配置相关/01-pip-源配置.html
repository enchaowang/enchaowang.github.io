      # 临时使用

      • linux/mac 创建或修改$HOME/.pip/pip.conf文件
      • windows 创建或修改$HOME/pip/pip.ini文件
      [global]
index-url = http://pypi.douban.com/simple
[install]
trusted-host = pypi.douban.com
